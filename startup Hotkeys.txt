Get_ID_List()
{
	WinGet, Title, List
	ret := []
	Loop, %Title%
	{
		temp_id := Title%A_Index%
		WinGetTitle, temp_title ,ahk_id %temp_id%
		if InStr(temp_title, "prodbass")
		{
			ret.Push(temp_id)
		}
	}
	Return ret
}


CreateLoadFile()
{
	FileDelete, H:\Settings_bass.txt

	temp_id := Get_ID_List()

	for i, id in temp_id
	{
		FileAppend, Bass%i%: %id%`r`n, H:\Settings_bass.txt
	}
	Return
}
